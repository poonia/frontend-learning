      /**
      * var declares a globle scope variable.
      */

      for (var i = 0; i < 10; i++) {
        console.log('var counter -> ' + i);
        setTimeout(function() {
          console.log('The var counter number after time out is -> ' + i);
        }, 1000);
      }





      /**
      * let declares a block scope local variable.
      */

      for (let i = 0; i < 10; i++) {
        console.log('let counter -> ' + i);
        setTimeout(function() {
          console.log('The let counter number after time out is -> ' + i);
        }, 1000);
      }
